// استبدل جزء التحقق في صفحة admin.html بهذا المنطق: async function handleLogin() { // ... جلب المدخلات ... // أولاً: التحقق في جدول saas_users (موظفو النظام) const { data: saasUser } = await _supabase .from('saas_users') .select('*') .eq('username', email) // هنا نستخدم اليوزر بدلاً من الإيميل أو كلاهما .eq('password', pass) .single(); if (saasUser) { // تحديث آخر ظهور await _supabase.from('saas_users').update({ last_login: new Date() }).eq('id', saasUser.id); sessionStorage.setItem('saasUser', JSON.stringify(saasUser)); window.location.href = 'saas-admin.html'; return; } // ثانياً: إذا لم يكن موظف نظام، ابحث في جدول staff (التجار) const { data: storeUser } = await _supabase .from('staff') .select('*') .eq('email', email) .eq('password', pass) .single(); if (storeUser) { sessionStorage.setItem('storeId', storeUser.store_id); window.location.href = `store-admin.html?store_id=${storeUser.store_id}`; } else { alert("بيانات الدخول خاطئة ❌"); } }
